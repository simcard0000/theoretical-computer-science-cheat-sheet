\Fm {n \choose k} = {n\over k}{n-1 \choose k-1}, \Mf

\Fm \sum_{k=0}^n {k \choose m} = { n+1 \choose m+1}, \Mf
\Fm \sum_{k=0}^n {r \choose k}{s \choose n -k} = {r+s \choose n}, \Mf
\Fm {n \choose k} =  (-1)^k {k-n-1 \choose k}, \Mf
